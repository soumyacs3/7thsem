// sr.c
#include &lt;mpi.h&gt;
#include &lt;stdio.h&gt;
#include &lt;string.h&gt;
int main(int argc, char** argv) {
int rank, size;
MPI_Status status;
// Initialize MPI
MPI_Init(&amp;argc, &amp;argv);
// Get current process rank and total number of processes
MPI_Comm_rank(MPI_COMM_WORLD, &amp;rank);
MPI_Comm_size(MPI_COMM_WORLD, &amp;size);
// Process 0 sends a message to all other processes
if (rank == 0) {
for (int i = 1; i &lt; size; i++) {
char message[100];
sprintf(message, &quot;Hello from process 0 to process %d&quot;, i);
MPI_Send(message, strlen(message) + 1, MPI_CHAR, i, 0, MPI_COMM_WORLD);
printf(&quot;Process 0 sent message to process %d\n&quot;, i);
}
} else {
// All other processes receive the message from process 0
char recv_buf[100];
MPI_Recv(recv_buf, 100, MPI_CHAR, 0, 0, MPI_COMM_WORLD, &amp;status);
printf(&quot;Process %d received message: %s\n&quot;, rank, recv_buf);
}
// Finalize MPI
MPI_Finalize();
return 0;
}
